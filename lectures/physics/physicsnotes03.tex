\bframe{Keeping at arm's length}
\begin{multicols}{2}
\includegraphics[scale=0.75]{Figures/lander01/lander_figure_05.png}

\columnbreak

\begin{eqnarray*}
A &=& p_2 - p_1\\
B &=& t - p_1\\
C &=& t - p_2\\
n &=& p_1 + A\frac{B\cdot A}{(B\cdot A) + (C \cdot A)}\\
\\
A_{norm} &=& \frac{A}{|A|}\\
n &=& p1 + A_{norm}(B\cdot A_{norm})
\end{eqnarray*}
\end{multicols}
